```
PROGRAM -> (STRUCT_DECL | INTERFACE_DECL | FUNC_DECL)* EOF
STRUCT_DECL -> "type" IDENTIFIER "struct" STRUCT_DEF_BLOCK
INTERFACE_DECL -> "type" IDENTIFIER "interface" INTERFACE_DEF_BLOCK
STRUCT_DEF_BLOCK  -> "{" (DECL_STMT)* "}"
INTERFACE_DEF_BLOCK -> "{" (METHOD_SIG EOL)* "}"
FUNC_DECL -> "func" METHOD_SIG BLOCK 
BLOCK -> "{" (STATEMENT)* "}" 
STATEMENT -> ASSIGN_STMT | EXPR_STMT | DECL_STMT | CONTROL_STMT | JUMP_STMT | RETURN_STMT | LOOP_STMT | BLOCK | EOL 
DECL_STMT -> "var"?  IDENTIFIER("," IDENTIFIER)* TYPE EOL
ASSIGN_STMT -> IDENTIFIER("," IDENTIFIER)* ("=" | ":=") EXPRESSION("," EXPRESSION)* EOL 
EXPR_STMT -> EXPRESSION EOL
CONTROL_STMT -> "if" EXPRESSION BLOCK ("else" (CONTROL_STMT | BLOCK))?
JUMP_STMT -> ("break" | "continue") EOL 
RETURN_STMT -> "return" (EXPRESSION(","EXPRESSION)*)? EOL
LOOP_STMT -> "for" EXPRESSION? BLOCK
CMPR_OP -> "==" | "!=" | ">" | ">=" | "<" | "<=" 
EXPRESSION -> CMPR_EXPR 
CMPR_EXPR -> ARITHMETIC (CMPR_OP ARITHMETIC)*
ARITHMETIC -> TERM(("+" | "-")TERM)*
TERM -> FACTOR(("*" | "/")FACTOR)* 
FACTOR -> IDENTIFIER ("++" | "--")? | FN_CALL | "(" EXPRESSION ")" | ERROR_LIT | LIT_VALS
LIT_VALS -> INT_LIT | FLOAT_LIT | STRING_LIT | BOOL_LIT
ERROR_LIT -> "error" "(" STRING_LIT ")"
BOOL_LIT -> "true" | "false"
TYPE -> "int" | "float" | "bool" | "string" | "err"
EOL -> "\n"
FN_CALL -> IDENTIFIER "(" (ARGS)? ")" 
ARGS -> EXPRESSION ("," EXPRESSION)*
PARAM -> (IDENTIFIER TYPE)
METHOD_SIG -> IDENTIFIER "(" (PARAM ("," PARAM)*)? ")" RETURN_TYPES?
RETURN_TYPES -> TYPE | "(" TYPE ("," TYPE)+ ")"
```
