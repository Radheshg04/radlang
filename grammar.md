PROGRAM -> (FUNC_DECL)* EOF 
FUNC_DECL -> "func" IDENTIFIER "(" ")" BLOCK 
BLOCK -> "{" STATEMENT_GROUP "}" 
STATEMENT_GROUP -> (STATEMENT)* 
STATEMENT -> ASSIGN_STMT | EXPR_STMT | DECL_STMT | EOL 
DECL_STMT -> "var" IDENTIFIER TYPE EOL 
ASSIGN_STMT -> IDENTIFIER "=" EXPRESSION EOL 
EXPR_STMT -> IDENTIFIER "(" ")" EOL | IDENTIFIER "+" "+" EOL | IDENTIFIER "-" "-" EOL 
TYPE -> "int" | "float" | "bool" | "string" 
EXPRESSION -> TERM(("+" | "-")TERM)* | IDENTIFIER "(" ")" 
TERM -> FACTOR(("*" | "/")FACTOR)* 
FACTOR -> NUMBER | IDENTIFIER | "(" EXPRESSION ")" 
EOL -> "\n"
